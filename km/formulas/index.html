бһҡбһјбһ”бһҳбһ“бҹ’бһҸбҹ–

DPS = бһ…бҹҶбһ“бһҪбһ“бһ…бһҳбҹ’бһ„бһ¶бһҷ / бһҠбһ¶бһ…бҹӢбһҹбһҡбһҹбҹғбһҲбһ¶бһҳбһҒбһҪбһҡбһҖбҹ’бһ”бһ¶бһӣ
DPS = бһ…бһҳбҹ’бһ„бһ¶бһҷ / (SR / 60)

бһҖбһ¶бһҡбһўбһ“бһ»бһңбһҸбҹ’бһҸ JavaScriptбҹ–

}

}

бһҳбһ»бһҒбһ„бһ¶бһҡбһңбһ·бһ—бһ¶бһӮPaceConsistency(laps) {
  const paces = laps.map(lap => lap.distance / lap.time);
  const avgPace = paces.reduce((a, b) => a + b) / paces.length;

  const variance = paces.reduce((sum, pace) =>
    бһ•бһӣбһ”бһјбһҖ + Math.pow(pace - avgPace, 2), 0) / paces.length;
  const stdDev = Math.sqrt(бһңбҹүбһ¶бһҡбҹ’бһҷбһ„бҹӢ);
  const coefficientOfVariation = (stdDev / avgPace) * 100;

  бһҸбҹ’бһҡбһЎбһ”бҹӢ {
    avgPace,
    stdDev,
    бһҳбҹҒбһӮбһ»бһҺбһ”бҹҶбһҡбҹӮбһ”бҹҶбһҡбһҪбһӣ,
    бһ—бһ¶бһ–бһҮбһ¶бһ”бҹӢбһӣбһ¶бһ”бҹӢбҹ– coefficientOfVariation < 5 ? "бһӣбҹ’бһўбһҘбһҸбһҒбҹ’бһ…бҹ„бҹҮ":
                 coefficientOfVariation < 10 ? "бһӣбҹ’бһў":
                 coefficientOfVariation < 15 ? "бһҳбһ’бҹ’бһҷбһҳ": "бһўбһҗбҹҒбһҡ"
  };
}
