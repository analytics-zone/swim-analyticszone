தொழில்நுட்ப குறிப்பு & சூத்திரங்கள்

முழுமையான கணித செயல்படுத்தல்

செயல்படுத்தல் வழிகாட்டி

இந்த பக்கம் அனைத்து நீச்சல் பகுப்பாய்வு அளவீடுகளுக்கும் நகல்-ஒட்டு சூத்திரங்கள் மற்றும் படிப்படியான கணக்கீட்டு முறைகளை வழங்குகிறது. தனிப்பயன் செயல்படுத்தல்கள், சரிபார்ப்பு அல்லது ஆழமான புரிதலுக்கு இவற்றை பயன்படுத்தவும்.

⚠️ செயல்படுத்தல் குறிப்புகள்

  • அனைத்து நேரங்களும் கணக்கீடுகளுக்கு வினாடிகளாக மாற்றப்பட வேண்டும்
  • நீச்சல் வேகம் தலைகீழ் (அதிக % = மெதுவான வேகம்)
  • நியாயமான வரம்புகளுக்கு உள்ளீடுகளை எப்போதும் சரிபார்க்கவும்
  • எட்ஜ் கேஸ்களை கையாளுங்கள் (பூஜ்ஜியத்தால் வகுத்தல், எதிர்மறை மதிப்புகள்)

முக்கிய செயல்திறன் அளவீடுகள்

முக்கிய நீச்சல் வேகம் (CSS)

சூத்திரம்:

CSS (m/s) = (D₂ - D₁) / (T₂ - T₁)
CSS வேகம்/100m (வினாடிகள்) = (T₄₀₀ - T₂₀₀) / 2

JavaScript செயல்படுத்தல்:

function calculateCSS(distance1, time1, distance2, time2) {
  // தேவைப்பட்டால் நேரங்களை வினாடிகளாக மாற்றவும்
  const t1 = typeof time1 === 'string' ? timeToSeconds(time1) : time1;
  const t2 = typeof time2 === 'string' ? timeToSeconds(time2) : time2;

  // CSS ஐ m/s இல் கணக்கிடவும்
  const css_ms = (distance2 - distance1) / (t2 - t1);

  // 100m க்கு வேகத்தை வினாடிகளில் கணக்கிடவும்
  const pace_per_100m = 100 / css_ms;

  // mm:ss வடிவத்திற்கு மாற்றவும்
  const minutes = Math.floor(pace_per_100m / 60);
  const seconds = Math.round(pace_per_100m % 60);

  return {
    css_ms: css_ms,
    pace_seconds: pace_per_100m,
    pace_formatted: `${minutes}:${seconds.toString().padStart(2, '0')}`
  };
}

// உதாரண பயன்பாடு:
const result = calculateCSS(200, 150, 400, 368);
// திரும்புகிறது: { css_ms: 0.917, pace_seconds: 109, pace_formatted: "1:49" }

நீச்சல் பயிற்சி அழுத்த மதிப்பெண் (sTSS)

முழுமையான சூத்திரம்:

sTSS = (IF³) × கால அளவு (மணிநேரங்கள்) × 100
IF = NSS / FTP
NSS = மொத்த தூரம் / மொத்த நேரம் (m/min)

JavaScript செயல்படுத்தல்:

function calculateSTSS(distance, timeMinutes, ftpMetersPerMin) {
  // இயல்பாக்கப்பட்ட நீச்சல் வேகத்தை கணக்கிடவும்
  const nss = distance / timeMinutes;

  // தீவிரத்தன்மை காரணியை கணக்கிடவும்
  const intensityFactor = nss / ftpMetersPerMin;

  // மணிநேரங்களை கணக்கிடவும்
  const hours = timeMinutes / 60;

  // கனசதுர தீவிரத்தன்மை காரணியை பயன்படுத்தி sTSS கணக்கிடவும்
  const stss = Math.pow(intensityFactor, 3) * hours * 100;

  return Math.round(stss);
}

// உதாரண பயன்பாடு:
const stss = calculateSTSS(3000, 55, 64.5);
// திரும்புகிறது: 55

// உதவி: CSS ஐ FTP ஆக மாற்றவும்
function cssToFTP(cssPacePer100mSeconds) {
  // FTP m/min இல் = 100m / (நிமிடங்களில் வேகம்)
  return 100 / (cssPacePer100mSeconds / 60);
}

// உதாரணம்: 1:33 (93 வினாடிகள்) CSS
const ftp = cssToFTP(93); // திரும்புகிறது: 64.5 m/min

SWOLF

சூத்திரம்:

SWOLF = லேப் நேரம் (வினாடிகள்) + ஸ்ட்ரோக் எண்ணிக்கை
SWOLF₂₅ = (நேரம் × 25/குளம் நீளம்) + (ஸ்ட்ரோக்குகள் × 25/குளம் நீளம்)

JavaScript செயல்படுத்தல்:

function calculateSWOLF(timeSeconds, strokeCount) {
  return timeSeconds + strokeCount;
}

function calculateNormalizedSWOLF(timeSeconds, strokeCount, poolLength) {
  const normalizedTime = timeSeconds * (25 / poolLength);
  const normalizedStrokes = strokeCount * (25 / poolLength);
  return normalizedTime + normalizedStrokes;
}

// உதாரணம்:
const swolf = calculateSWOLF(20, 15);
// திரும்புகிறது: 35

const swolf50m = calculateNormalizedSWOLF(40, 30, 50);
// திரும்புகிறது: 35 (25m க்கு இயல்பாக்கப்பட்டது)

ஸ்ட்ரோக் இயக்கவியல்

ஸ்ட்ரோக் வீதம் (SR)

சூத்திரம்:

SR = 60 / சுழற்சி நேரம் (வினாடிகள்)
SR = (ஸ்ட்ரோக்குகளின் எண்ணிக்கை / வினாடிகளில் நேரம்) × 60

JavaScript செயல்படுத்தல்:

function calculateStrokeRate(strokeCount, timeSeconds) {
  return (strokeCount / timeSeconds) * 60;
}

// உதாரணம்:
const sr = calculateStrokeRate(30, 25);
// திரும்புகிறது: 72 SPM

ஸ்ட்ரோக் தூரம் (DPS)

சூத்திரம்:

DPS = தூரம் / ஸ்ட்ரோக் எண்ணிக்கை
DPS = வேகம் / (SR / 60)

JavaScript செயல்படுத்தல்:

function calculateDPS(distance, strokeCount, pushoffDistance = 0) {
  const effectiveDistance = distance - pushoffDistance;
  return effectiveDistance / strokeCount;
}

// உதாரணம் (25m குளம், 5m புஷ்-ஆஃப்):
const dps = calculateDPS(25, 12, 5);
// திரும்புகிறது: 1.67 m/stroke

// பல லேப்களுக்கு:
const dps100m = calculateDPS(100, 48, 4 * 5);
// திரும்புகிறது: 1.67 m/stroke (4 லேப்கள் × 5m புஷ்-ஆஃப்)

SR மற்றும் DPS இலிருந்து வேகம்

சூத்திரம்:

வேகம் (m/s) = (SR / 60) × DPS

JavaScript செயல்படுத்தல்:

function calculateVelocity(strokeRate, dps) {
  return (strokeRate / 60) * dps;
}

// உதாரணம்:
const velocity = calculateVelocity(70, 1.6);
// திரும்புகிறது: 1.87 m/s

ஸ்ட்ரோக் குறியீடு (SI)

சூத்திரம்:

SI = வேகம் (m/s) × DPS (m/stroke)

JavaScript செயல்படுத்தல்:

function calculateStrokeIndex(velocity, dps) {
  return velocity * dps;
}

// உதாரணம்:
const si = calculateStrokeIndex(1.5, 1.7);
// திரும்புகிறது: 2.55

செயல்திறன் மேலாண்மை விளக்கப்படம் (PMC)

CTL, ATL, TSB கணக்கீடுகள்

சூத்திரங்கள்:

CTL இன்று = CTL நேற்று + (TSS இன்று - CTL நேற்று) × (1/42)
ATL இன்று = ATL நேற்று + (TSS இன்று - ATL நேற்று) × (1/7)
TSB = CTL நேற்று - ATL நேற்று

JavaScript செயல்படுத்தல்:

function updateCTL(previousCTL, todayTSS) {
  return previousCTL + (todayTSS - previousCTL) * (1/42);
}

function updateATL(previousATL, todayTSS) {
  return previousATL + (todayTSS - previousATL) * (1/7);
}

function calculateTSB(yesterdayCTL, yesterdayATL) {
  return yesterdayCTL - yesterdayATL;
}

// பயிற்சிகளின் தொடருக்கு PMC கணக்கிடவும்
function calculatePMC(workouts) {
  let ctl = 0, atl = 0;
  const results = [];

  workouts.forEach(workout => {
    ctl = updateCTL(ctl, workout.tss);
    atl = updateATL(atl, workout.tss);
    const tsb = calculateTSB(ctl, atl);

    results.push({
      date: workout.date,
      tss: workout.tss,
      ctl: Math.round(ctl * 10) / 10,
      atl: Math.round(atl * 10) / 10,
      tsb: Math.round(tsb * 10) / 10
    });
  });

  return results;
}

// உதாரண பயன்பாடு:
const workouts = [
  { date: '2025-01-01', tss: 50 },
  { date: '2025-01-02', tss: 60 },
  { date: '2025-01-03', tss: 45 },
  // ... மேலும் பயிற்சிகள்
];

const pmc = calculatePMC(workouts);
// ஒவ்வொரு நாளுக்கும் CTL, ATL, TSB உடன் வரிசையை திரும்புகிறது

மேம்பட்ட கணக்கீடுகள்

பல தூரங்களிலிருந்து CSS (பின்னடைவு முறை)

JavaScript செயல்படுத்தல்:

function calculateCSSRegression(distances, times) {
  // நேரியல் பின்னடைவு: தூரம் = a + b*நேரம்
  const n = distances.length;
  const sumX = times.reduce((a, b) => a + b, 0);
  const sumY = distances.reduce((a, b) => a + b, 0);
  const sumXY = times.reduce((sum, x, i) => sum + x * distances[i], 0);
  const sumXX = times.reduce((sum, x) => sum + x * x, 0);

  const slope = (n * sumXY - sumX * sumY) / (n * sumXX - sumX * sumX);
  const intercept = (sumY - slope * sumX) / n;

  return {
    css: slope, // முக்கிய நீச்சல் வேகம் (m/s)
    anaerobic_capacity: intercept // காற்றில்லா தூர திறன் (m)
  };
}

// பல சோதனை தூரங்களுடன் உதாரணம்:
const distances = [100, 200, 400, 800];
const times = [65, 150, 340, 720]; // வினாடிகளில்
const result = calculateCSSRegression(distances, times);
// திரும்புகிறது: { css: 1.18, anaerobic_capacity: 15.3 }

வேகத்திலிருந்து தீவிரத்தன்மை காரணி

JavaScript செயல்படுத்தல்:

function calculateIntensityFactor(actualPace100m, thresholdPace100m) {
  // வேகத்தை வேகமாக மாற்றவும் (m/s)
  const actualSpeed = 100 / actualPace100m;
  const thresholdSpeed = 100 / thresholdPace100m;
  return actualSpeed / thresholdSpeed;
}

// உதாரணம்:
const if_value = calculateIntensityFactor(110, 93);
// திரும்புகிறது: 0.845 (வாசலின் 84.5% இல் நீச்சல்)

வேக நிலைத்தன்மை பகுப்பாய்வு

JavaScript செயல்படுத்தல்:

function analyzePaceConsistency(laps) {
  const paces = laps.map(lap => lap.distance / lap.time);
  const avgPace = paces.reduce((a, b) => a + b) / paces.length;

  const variance = paces.reduce((sum, pace) =>
    sum + Math.pow(pace - avgPace, 2), 0) / paces.length;
  const stdDev = Math.sqrt(variance);
  const coefficientOfVariation = (stdDev / avgPace) * 100;

  return {
    avgPace,
    stdDev,
    coefficientOfVariation,
    consistency: coefficientOfVariation < 5 ? "சிறந்தது" :
                 coefficientOfVariation < 10 ? "நல்லது" :
                 coefficientOfVariation < 15 ? "மிதமானது" : "மாறுபட்டது"
  };
}

// உதாரணம்:
const laps = [
  { distance: 100, time: 70 },
  { distance: 100, time: 72 },
  { distance: 100, time: 71 },
  // ...
];
const analysis = analyzePaceConsistency(laps);
// திரும்புகிறது: { avgPace: 1.41, stdDev: 0.02, coefficientOfVariation: 1.4, consistency: "சிறந்தது" }

ஸ்ட்ரோக் எண்ணிக்கையிலிருந்து சோர்வு கண்டறிதல்

JavaScript செயல்படுத்தல்:

function detectFatigue(laps) {
  const firstThird = laps.slice(0, Math.floor(laps.length/3));
  const lastThird = laps.slice(-Math.floor(laps.length/3));

  const firstThirdAvg = firstThird.reduce((sum, lap) =>
    sum + lap.strokeCount, 0) / firstThird.length;
  const lastThirdAvg = lastThird.reduce((sum, lap) =>
    sum + lap.strokeCount, 0) / lastThird.length;

  const strokeCountIncrease = ((lastThirdAvg - firstThirdAvg) / firstThirdAvg) * 100;

  return {
    firstThirdAvg: Math.round(firstThirdAvg * 10) / 10,
    lastThirdAvg: Math.round(lastThirdAvg * 10) / 10,
    percentIncrease: Math.round(strokeCountIncrease * 10) / 10,
    fatigueLevel: strokeCountIncrease < 5 ? "குறைந்தபட்சம்" :
                  strokeCountIncrease < 10 ? "மிதமானது" :
                  strokeCountIncrease < 20 ? "குறிப்பிடத்தக்கது" : "கடுமையானது"
  };
}

// உதாரணம்:
const laps = [
  { strokeCount: 14 }, { strokeCount: 14 }, { strokeCount: 15 },
  { strokeCount: 15 }, { strokeCount: 16 }, { strokeCount: 16 },
  { strokeCount: 17 }, { strokeCount: 18 }, { strokeCount: 18 }
];
const fatigue = detectFatigue(laps);
// திரும்புகிறது: { firstThirdAvg: 14.3, lastThirdAvg: 17.7, percentIncrease: 23.8, fatigueLevel: "கடுமையானது" }

தரவு சரிபார்ப்பு

பயிற்சி தரவு தர சோதனைகள்

JavaScript செயல்படுத்தல்:

function validateWorkoutData(workout) {
  const issues = [];

  // நியாயமான வேக வரம்புகளை சோதிக்கவும் (100m க்கு 1:00-5:00)
  const avgPace = (workout.totalTime / workout.totalDistance) * 100;
  if (avgPace < 60 || avgPace > 300) {
    issues.push(`அசாதாரண சராசரி வேகம்: 100m க்கு ${Math.round(avgPace)}s`);
  }

  // நியாயமான ஸ்ட்ரோக் எண்ணிக்கைகளை சோதிக்கவும் (25m க்கு 10-50)
  const avgStrokesPer25m = (workout.totalStrokes / workout.totalDistance) * 25;
  if (avgStrokesPer25m < 10 || avgStrokesPer25m > 50) {
    issues.push(`அசாதாரண ஸ்ட்ரோக் எண்ணிக்கை: 25m க்கு ${Math.round(avgStrokesPer25m)}`);
  }

  // நியாயமான ஸ்ட்ரோக் வீதத்தை சோதிக்கவும் (30-150 SPM)
  const avgSR = calculateStrokeRate(workout.totalStrokes, workout.totalTime);
  if (avgSR < 30 || avgSR > 150) {
    issues.push(`அசாதாரண ஸ்ட்ரோக் வீதம்: ${Math.round(avgSR)} SPM`);
  }

  // விடுபட்ட லேப்களை சோதிக்கவும் (நேரத்தில் இடைவெளிகள்)
  if (workout.laps && workout.laps.length > 1) {
    for (let i = 1; i < workout.laps.length; i++) {
      const gap = workout.laps[i].startTime -
                  (workout.laps[i-1].startTime + workout.laps[i-1].duration);
      if (gap > 300) { // 5 நிமிட இடைவெளி
        issues.push(`லேப்கள் ${i} மற்றும் ${i+1} இடையே பெரிய இடைவெளி கண்டறியப்பட்டது`);
      }
    }
  }

  return {
    isValid: issues.length === 0,
    issues
  };
}

// உதாரணம்:
const workout = {
  totalDistance: 2000,
  totalTime: 1800, // 30 நிமிடங்கள்
  totalStrokes: 800,
  laps: [/* லேப் தரவு */]
};
const validation = validateWorkoutData(workout);
// திரும்புகிறது: { isValid: true, issues: [] }

நேர மாற்ற பயன்பாடுகள்

JavaScript செயல்படுத்தல்:

// mm:ss ஐ வினாடிகளாக மாற்றவும்
function timeToSeconds(timeString) {
  const parts = timeString.split(':');
  return parseInt(parts[0]) * 60 + parseInt(parts[1]);
}

// வினாடிகளை mm:ss ஆக மாற்றவும்
function secondsToTime(seconds) {
  const minutes = Math.floor(seconds / 60);
  const secs = Math.round(seconds % 60);
  return `${minutes}:${secs.toString().padStart(2, '0')}`;
}

// வினாடிகளை hh:mm:ss ஆக மாற்றவும்
function secondsToTimeDetailed(seconds) {
  const hours = Math.floor(seconds / 3600);
  const minutes = Math.floor((seconds % 3600) / 60);
  const secs = Math.round(seconds % 60);
  return `${hours}:${minutes.toString().padStart(2, '0')}:${secs.toString().padStart(2, '0')}`;
}

// உதாரணங்கள்:
timeToSeconds("1:33"); // திரும்புகிறது: 93
secondsToTime(93); // திரும்புகிறது: "1:33"
secondsToTimeDetailed(3665); // திரும்புகிறது: "1:01:05"

செயல்படுத்தல் வளங்கள்

இந்த பக்கத்தில் உள்ள அனைத்து சூத்திரங்களும் உற்பத்தி-தயாராக உள்ளன மற்றும் அறிவியல் இலக்கியத்திற்கு எதிராக சரிபார்க்கப்பட்டுள்ளன. தனிப்பயன் பகுப்பாய்வு கருவிகள், சரிபார்ப்பு அல்லது நீச்சல் செயல்திறன் கணக்கீடுகளின் ஆழமான புரிதலுக்கு இவற்றை பயன்படுத்தவும்.

💡 சிறந்த நடைமுறைகள்

  • உள்ளீடுகளை சரிபார்க்கவும்: கணக்கிடுவதற்கு முன் நியாயமான வரம்புகளை சோதிக்கவும்
  • எட்ஜ் கேஸ்களை கையாளுங்கள்: பூஜ்ஜியத்தால் வகுத்தல், எதிர்மறை மதிப்புகள், null தரவு
  • சரியாக வட்டமிடுங்கள்: CTL/ATL/TSB ஐ 1 தசமத்திற்கு, sTSS ஐ முழு எண்ணாக
  • துல்லியத்தை சேமிக்கவும்: தரவுத்தளத்தில் முழு துல்லியத்தை வைத்திருங்கள், காட்சிக்கு வட்டமிடுங்கள்
  • முழுமையாக சோதிக்கவும்: கணக்கீடுகளை சரிபார்க்க அறியப்பட்ட-நல்ல தரவை பயன்படுத்தவும்
